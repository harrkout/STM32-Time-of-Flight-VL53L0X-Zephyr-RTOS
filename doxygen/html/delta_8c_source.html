15static int erase_page(struct flash_mem* flash, off_t offset)
16{
17 offset = offset - offset % PAGE_SIZE; /* find start of page */
18
19 if (flash_write_protection_set(flash->device, false)) { return -DELTA_CLEARING_ERROR; }
20 if (flash_erase(flash->device, offset, PAGE_SIZE)) { return -DELTA_CLEARING_ERROR; }
21 if (flash_write_protection_set(flash->device, true)) { return -DELTA_CLEARING_ERROR; }
22
23 return DELTA_OK;
24}

299static int delta_flash_from_read(void* arg_p, uint8_t* buf_p, size_t size)
300{
301 struct flash_mem* flash;
302
303 flash = (struct flash_mem*)arg_p;
304
305 if (!flash) { return -DELTA_CASTING_ERROR; }
306 if (size <= 0) { return -DELTA_INVALID_BUF_SIZE; }
307
308 if (flash_read(flash->device, flash->from_current, buf_p, size)) { return -DELTA_READING_SOURCE_ERROR; }
309
310 flash->from_current += (off_t)size;
311
312 printk("\n-----delta_flash_from_read-----\n\nflash->from_current: 0x%08x \n", flash->from_current);
313
314 if (flash->from_current >= flash->from_end) { return -DELTA_READING_SOURCE_ERROR; }
315
316 return DELTA_OK;
317}

363{
364 if (!flash) { return -DELTA_NO_FLASH_FOUND; }
365
366 flash->from_current = PRIMARY_OFFSET;

563const char* delta_error_as_string(int error)
564{
565 if (error < 28) { return detools_error_as_string(error); }
566
567 if (error < 0) { error *= -1; }
568
569 switch (error) {
570 case DELTA_SLOT1_OUT_OF_MEMORY:
571 return "Slot 1 out of memory.";
572 case DELTA_READING_PATCH_ERROR:
573 return "Error reading patch.";
574 case DELTA_READING_SOURCE_ERROR:
575 return "Error reading source image.";
576 case DELTA_WRITING_ERROR:
577 return "Error writing to slot 1.";
578 case DELTA_SEEKING_ERROR:
579 return "Seek error.";
580 case DELTA_CASTING_ERROR:
581 return "Error casting to flash_mem.";
582 case DELTA_INVALID_BUF_SIZE:
583 return "Read/write buffer less or equal to 0.";
584 case DELTA_CLEARING_ERROR:
585 return "Could not clear slot 1.";
586 case DELTA_NO_FLASH_FOUND:
587 return "No flash found.";
588 case DELTA_PATCH_HEADER_ERROR:
589 return "Error reading patch header.";
590 default:
591 return "Unknown error.";
592 }
