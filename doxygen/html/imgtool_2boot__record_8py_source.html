37def create_sw_component_data(sw_type, sw_version, sw_measurement_description,
38 sw_measurement_value, sw_signer_id):
39
40 # List of software component properties (Key ID + value)
41 properties = {
42 SwComponent.TYPE: sw_type,

44 SwComponent.SIGNER_ID: sw_signer_id,
45 SwComponent.MEASUREMENT_DESCRIPTION: sw_measurement_description,
