288static void *temperature_get_buf(uint16_t obj_inst_id, uint16_t res_id, uint16_t res_inst_id,
289 size_t *data_len)
290{
291 /* Last read temperature value, will use 25.5C if no sensor available */
292 static double v = 25.5;
293 const struct device *dev = NULL;
294
295#if defined(CONFIG_FXOS8700_TEMP)
296 dev = device_get_binding(DT_LABEL(DT_INST(0, nxp_fxos8700)));
297#endif
298
299 if (dev != NULL) {
300 struct sensor_value val;
301
302 if (sensor_sample_fetch(dev)) {
303 LOG_ERR("temperature data update failed");
304 }
305
306 sensor_channel_get(dev, SENSOR_CHAN_DIE_TEMP, &val);
307
308 v = sensor_value_to_double(&val);
309
310 LOG_DBG("LWM2M temperature set to %f", v);
311 }
312
313 /* echo the value back through the engine to update min/max values */
314 lwm2m_engine_set_float("3303/0/5700", &v);
315 *data_len = sizeof(v);
316 return &v;
317}

369static int lwm2m_setup(void)
370{
371 int ret;
372 char *server_url;
373 uint16_t server_url_len;
374 uint8_t server_url_flags;
375
376 /* setup SECURITY object */
377
378 /* Server URL */
379 ret = lwm2m_engine_get_res_data("0/0/0", (void **)&server_url, &server_url_len,
380 &server_url_flags);
381 if (ret < 0) {
382 return ret;
383 }
384
385 snprintk(server_url, server_url_len, "coap%s//%s%s%s",
386 IS_ENABLED(CONFIG_LWM2M_DTLS_SUPPORT) ? "s:" : ":",
387 strchr(SERVER_ADDR, ':') ? "[" : "", SERVER_ADDR,
388 strchr(SERVER_ADDR, ':') ? "]" : "");
389
390 /* Security Mode */
391 lwm2m_engine_set_u8("0/0/2", IS_ENABLED(CONFIG_LWM2M_DTLS_SUPPORT) ? 0 : 3);
392#if defined(CONFIG_LWM2M_DTLS_SUPPORT)
393 lwm2m_engine_set_string("0/0/0", "coap://10.0.2.170:5683");
394 lwm2m_engine_set_string("0/0/3", (char *)client_psk_id);
395 lwm2m_engine_set_opaque("0/0/5", (void *)client_psk, sizeof(client_psk));
396#endif /* CONFIG_LWM2M_DTLS_SUPPORT */
397
398#if defined(CONFIG_LWM2M_RD_CLIENT_SUPPORT_BOOTSTRAP)
399 /* Mark 1st instance of security object as a bootstrap server */
400 lwm2m_engine_set_u8("0/0/1", 1);
401
402 /* Create 2nd instance of security object needed for bootstrap */
403 lwm2m_engine_create_obj_inst("0/1");
404#else
405 /* Match Security object instance with a Server object instance with
406 * Short Server ID.
407 */
408 lwm2m_engine_set_u16("0/0/10", 101);
409 lwm2m_engine_set_u16("1/0/0", 101);
410#endif
411
412 /* setup SERVER object */
413
414 /* setup DEVICE object */
415
416 lwm2m_engine_set_res_data("3/0/0", CLIENT_MANUFACTURER, sizeof(CLIENT_MANUFACTURER),
417 LWM2M_RES_DATA_FLAG_RO);
418 lwm2m_engine_set_res_data("3/0/1", CLIENT_MODEL_NUMBER, sizeof(CLIENT_MODEL_NUMBER),
419 LWM2M_RES_DATA_FLAG_RO);
420 lwm2m_engine_set_res_data("3/0/2", CLIENT_SERIAL_NUMBER, sizeof(CLIENT_SERIAL_NUMBER),
421 LWM2M_RES_DATA_FLAG_RO);
422 lwm2m_engine_set_res_data("3/0/3", CLIENT_FIRMWARE_VER, sizeof(CLIENT_FIRMWARE_VER),
423 LWM2M_RES_DATA_FLAG_RO);
424 lwm2m_engine_register_exec_callback("3/0/4", device_reboot_cb);
425 lwm2m_engine_register_exec_callback("3/0/5", device_factory_default_cb);
426 lwm2m_engine_set_res_data("3/0/9", &bat_level, sizeof(bat_level), 0);
427 lwm2m_engine_set_res_data("3/0/10", &mem_free, sizeof(mem_free), 0);
428 lwm2m_engine_set_res_data("3/0/17", CLIENT_DEVICE_TYPE, sizeof(CLIENT_DEVICE_TYPE),
429 LWM2M_RES_DATA_FLAG_RO);
430 lwm2m_engine_set_res_data("3/0/18", CLIENT_HW_VER, sizeof(CLIENT_HW_VER),
431 LWM2M_RES_DATA_FLAG_RO);
432 lwm2m_engine_set_res_data("3/0/20", &bat_status, sizeof(bat_status), 0);
433 lwm2m_engine_set_res_data("3/0/21", &mem_total, sizeof(mem_total), 0);
434
435 /* add power source resource instances */
436 lwm2m_engine_create_res_inst("3/0/6/0");
437 lwm2m_engine_set_res_data("3/0/6/0", &bat_idx, sizeof(bat_idx), 0);
438
439 lwm2m_engine_create_res_inst("3/0/7/0");
440 lwm2m_engine_set_res_data("3/0/7/0", &bat_mv, sizeof(bat_mv), 0);
441 lwm2m_engine_create_res_inst("3/0/8/0");
442 lwm2m_engine_set_res_data("3/0/8/0", &bat_ma, sizeof(bat_ma), 0);
443 lwm2m_engine_create_res_inst("3/0/6/1");
444 lwm2m_engine_set_res_data("3/0/6/1", &usb_idx, sizeof(usb_idx), 0);
445 lwm2m_engine_create_res_inst("3/0/7/1");
446 lwm2m_engine_set_res_data("3/0/7/1", &usb_mv, sizeof(usb_mv), 0);
447 lwm2m_engine_create_res_inst("3/0/8/1");
448 lwm2m_engine_set_res_data("3/0/8/1", &usb_ma, sizeof(usb_ma), 0);
449
450 /* setup FIRMWARE object */
451
452#if defined(CONFIG_LWM2M_FIRMWARE_UPDATE_OBJ_SUPPORT)
453 /* setup data buffer for block-wise transfer */
454 lwm2m_engine_register_pre_write_callback("5/0/0", firmware_get_buf);
455 lwm2m_firmware_set_write_cb(firmware_block_received_cb);
456#endif
457#if defined(CONFIG_LWM2M_FIRMWARE_UPDATE_PULL_SUPPORT)
458 lwm2m_engine_create_res_inst("5/0/8/0");
459 lwm2m_engine_set_res_data("5/0/8/0", &supported_protocol[0], sizeof(supported_protocol[0]),
460 0);
461
462 lwm2m_firmware_set_update_cb(firmware_update_cb);
463#endif
464
465 /* setup TEMP SENSOR object */
466 lwm2m_engine_create_obj_inst("3303/0");
467 lwm2m_engine_register_read_callback("3303/0/5700", temperature_get_buf);
468
469 /* IPSO: Light Control object */
470 if (init_led_device() == 0) {
471 lwm2m_engine_create_obj_inst("3311/0");
472 lwm2m_engine_register_post_write_callback("3311/0/5850", led_on_off_cb);
473 lwm2m_engine_set_res_data("3311/0/5750", LIGHT_NAME, sizeof(LIGHT_NAME),
474 LWM2M_RES_DATA_FLAG_RO);
475 }
476
477 /* IPSO: Timer object */
478 lwm2m_engine_create_obj_inst("3340/0");
479 lwm2m_engine_register_validate_callback("3340/0/5850", timer_on_off_validate_cb);
480 lwm2m_engine_register_post_write_callback("3340/0/5543", timer_digital_state_cb);
481 lwm2m_engine_set_res_data("3340/0/5750", TIMER_NAME, sizeof(TIMER_NAME),
482 LWM2M_RES_DATA_FLAG_RO);
483
484 return 0;
485}

487static void rd_client_event(struct lwm2m_ctx *client, enum lwm2m_rd_client_event client_event)
488{
489 switch (client_event) {
490 case LWM2M_RD_CLIENT_EVENT_NONE:
491 /* do nothing */
492 break;
493
494 case LWM2M_RD_CLIENT_EVENT_BOOTSTRAP_REG_FAILURE:
495 LOG_DBG("Bootstrap registration failure!");
496 break;
497
498 case LWM2M_RD_CLIENT_EVENT_BOOTSTRAP_REG_COMPLETE:
499 LOG_DBG("Bootstrap registration complete");
500 break;
501
502 case LWM2M_RD_CLIENT_EVENT_BOOTSTRAP_TRANSFER_COMPLETE:
503 LOG_DBG("Bootstrap transfer complete");
504 break;
505
506 case LWM2M_RD_CLIENT_EVENT_REGISTRATION_FAILURE:
507 LOG_DBG("Registration failure!");
508 break;
509
510 case LWM2M_RD_CLIENT_EVENT_REGISTRATION_COMPLETE:
511 LOG_DBG("Registration complete");
512 break;
513
514 case LWM2M_RD_CLIENT_EVENT_REG_UPDATE_FAILURE:
515 LOG_DBG("Registration update failure!");
516 break;
517
518 case LWM2M_RD_CLIENT_EVENT_REG_UPDATE_COMPLETE:
519 LOG_DBG("Registration update complete");
520 break;
521
522 case LWM2M_RD_CLIENT_EVENT_DEREGISTER_FAILURE:
523 LOG_DBG("Deregister failure!");
524 break;
525
526 case LWM2M_RD_CLIENT_EVENT_DISCONNECT:
527 LOG_DBG("Disconnected");
528 break;
529
530 case LWM2M_RD_CLIENT_EVENT_QUEUE_MODE_RX_OFF:
531 LOG_DBG("Queue mode RX window closed");
532 break;
533
534 case LWM2M_RD_CLIENT_EVENT_NETWORK_ERROR:
535 LOG_ERR("LwM2M engine reported a network erorr.");
536 lwm2m_rd_client_stop(client, rd_client_event, true);
537 break;
538 }
539}

541static void observe_cb(enum lwm2m_observe_event event, struct lwm2m_obj_path *path, void *user_data)
542{
543 char buf[LWM2M_MAX_PATH_STR_LEN];
544
545 switch (event) {
546 case LWM2M_OBSERVE_EVENT_OBSERVER_ADDED:
547 LOG_INF("Observer added for %s", lwm2m_path_log_strdup(buf, path));
548 break;
549
550 case LWM2M_OBSERVE_EVENT_OBSERVER_REMOVED:
551 LOG_INF("Observer removed for %s", lwm2m_path_log_strdup(buf, path));
552 break;
553
554 case LWM2M_OBSERVE_EVENT_NOTIFY_ACK:
555 LOG_INF("Notify acknowledged for %s", lwm2m_path_log_strdup(buf, path));
556 break;
557
558 case LWM2M_OBSERVE_EVENT_NOTIFY_TIMEOUT:
559 LOG_INF("Notify timeout for %s, trying registration update",
560 lwm2m_path_log_strdup(buf, path));
561
562 lwm2m_rd_client_update();
563 break;
564 }

757{

767 gpio_add_callback(button, &button_cb_data);
768
769 btn_flag = false;
770
771 return 0;
772}
