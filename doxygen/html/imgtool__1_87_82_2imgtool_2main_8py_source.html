138@click.argument('imgfile')
139@click.option('-k', '--key', metavar='filename')
140@click.command(help="Check that signed image can be verified by given key")
141def verify(key, imgfile):
142 key = load_key(key) if key else None
143 ret, version, digest = image.Image.verify(imgfile, key)
144 if ret == image.VerifyResult.OK:
145 print("Image was correctly validated")
146 print("Image version: {}.{}.{}+{}".format(*version))
147 print("Image digest: {}".format(digest.hex()))
148 return
149 elif ret == image.VerifyResult.INVALID_MAGIC:
150 print("Invalid image magic; is this an MCUboot image?")
151 elif ret == image.VerifyResult.INVALID_TLV_INFO_MAGIC:
152 print("Invalid TLV info magic; is this an MCUboot image?")
153 elif ret == image.VerifyResult.INVALID_HASH:
154 print("Image has an invalid sha256 digest")
155 elif ret == image.VerifyResult.INVALID_SIGNATURE:
156 print("No signature found for the given key")
157 else:
158 print("Unknown return code: {}".format(ret))
159 sys.exit(1)

192def get_dependencies(ctx, param, value):
193 if value is not None:
194 versions = []
195 images = re.findall(r"\‍((\d+)", value)
196 if len(images) == 0:
197 raise click.BadParameter(
198 "Image dependency format is invalid: {}".format(value))
199 raw_versions = re.findall(r",\s*([0-9.+]+)\‍)", value)
200 if len(images) != len(raw_versions):
201 raise click.BadParameter(
202 '''There's a mismatch between the number of dependency images
203 and versions in: {}'''.format(value))
204 for raw_version in raw_versions:
205 try:
206 versions.append(decode_version(raw_version))
207 except ValueError as e:
208 raise click.BadParameter("{}".format(e))
209 dependencies = dict()
210 dependencies[image.DEP_IMAGES_KEY] = images
211 dependencies[image.DEP_VERSIONS_KEY] = versions
212 return dependencies

294def sign(key, public_key_format, align, version, pad_sig, header_size,
295 pad_header, slot_size, pad, confirm, max_sectors, overwrite_only,
296 endian, encrypt, infile, outfile, dependencies, load_addr, hex_addr,
297 erased_val, save_enctlv, security_counter, boot_record, custom_tlv,
298 rom_fixed):
299
300 if confirm:
301 # Confirmed but non-padded images don't make much sense, because
302 # otherwise there's no trailer area for writing the confirmed status.
303 pad = True
304 img = image.Image(version=decode_version(version), header_size=header_size,
305 pad_header=pad_header, pad=pad, confirm=confirm,
306 align=int(align), slot_size=slot_size,
307 max_sectors=max_sectors, overwrite_only=overwrite_only,
308 endian=endian, load_addr=load_addr, rom_fixed=rom_fixed,
309 erased_val=erased_val, save_enctlv=save_enctlv,
310 security_counter=security_counter)
311 img.load(infile)
312 key = load_key(key) if key else None
313 enckey = load_key(encrypt) if encrypt else None
314 if enckey and key:
315 if ((isinstance(key, keys.ECDSA256P1) and
316 not isinstance(enckey, keys.ECDSA256P1Public))
317 or (isinstance(key, keys.RSA) and
318 not isinstance(enckey, keys.RSAPublic))):
319 # FIXME
320 raise click.UsageError("Signing and encryption must use the same "
321 "type of key")
322
323 if pad_sig and hasattr(key, 'pad_sig'):
324 key.pad_sig = True
325
326 # Get list of custom protected TLVs from the command-line
327 custom_tlvs = {}
328 for tlv in custom_tlv:
329 tag = int(tlv[0], 0)
330 if tag in custom_tlvs:
331 raise click.UsageError('Custom TLV %s already exists.' % hex(tag))
332 if tag in image.TLV_VALUES.values():
333 raise click.UsageError(
334 'Custom TLV %s conflicts with predefined TLV.' % hex(tag))
335
336 value = tlv[1]
337 if value.startswith('0x'):
338 if len(value[2:]) % 2:
339 raise click.UsageError('Custom TLV length is odd.')
340 custom_tlvs[tag] = bytes.fromhex(value[2:])
341 else:
342 custom_tlvs[tag] = value.encode('utf-8')
343
344 img.create(key, public_key_format, enckey, dependencies, boot_record,
345 custom_tlvs)
346 img.save(outfile, hex_addr)
